DECLARE ROWS: INTEGER
DECLARE COLS: INTEGER
DECLARE I: INTEGER
DECLARE J: INTEGER
DECLARE ARR[100][100]: INTEGER
DECLARE SUM: INTEGER

READ ROWS
READ COLS

FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        READ ARR[I][J]
    END FOR
END FOR

SUM := 0
FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        SUM := SUM + ARR[I][J]
    END FOR
END FOR

PRINT SUM
