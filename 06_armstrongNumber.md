DECLARE N: INTEGER
DECLARE TEMP: INTEGER
DECLARE DIGIT: INTEGER
DECLARE SUM: INTEGER

READ N
TEMP := N
SUM := 0

WHILE TEMP > 0
    DIGIT := TEMP MOD 10
    SUM := SUM + DIGIT * DIGIT * DIGIT
    TEMP := TEMP / 10
END WHILE

IF SUM = N
    PRINT "Armstrong Number"
ELSE
    PRINT "Not Armstrong Number"
END IF
