DECLARE ROWS: INTEGER
DECLARE COLS: INTEGER
DECLARE I: INTEGER
DECLARE J: INTEGER
DECLARE A[100][100]: INTEGER
DECLARE B[100][100]: INTEGER
DECLARE SUM[100][100]: INTEGER

READ ROWS
READ COLS

FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        READ A[I][J]
    END FOR
END FOR

FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        READ B[I][J]
    END FOR
END FOR

FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        SUM[I][J] := A[I][J] + B[I][J]
    END FOR
END FOR

FOR I := 0 TO ROWS-1
    FOR J := 0 TO COLS-1
        PRINT SUM[I][J]
    END FOR
END FOR
